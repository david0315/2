    1.实例化Class类对象

    //第一种，可靠高效
Class c0 = String.class ;
//第二种，使用对象
Class c1 = "zhangsan".getClass() ;
//第三种，常用，可能会抛出ClassNotFoundException
Class c2 = Class.forName("java.lang.String") ;
//第四种，不常用
ClassLoader cl = this.getClass().getClassLoader() ;
Class c3 = cl.loadClass("类的全类名") ;
