    1.创建string的常用3+1种方式

    /*
创建string的常用3+1种方式
三种构造方法
public String()：创建一个空字符串，不含有任何内容；
public String(char[] array)：根据字符数组的内容，来创建对应的字符串。
public String(byte[] array)：根据字节数组的内容，来创建对应的字符串。
一种直接创建：
String str = "Hello";//右边直接使用双引号

注意：直接写上双引号，就是字符串对象。
*/
public class demo01String {
    public static void main(String[] args) {
        String str1 = new String();//小括号留空说明字符串什么内容都没有
        System.out.println("第一个字符串：" + str1);
        //根据字符数组创建字符串
        char[] charArray = {'A', 'B', 'C'};
        String str2 = new String(charArray);
        System.out.println("第二个字符串：" + str2);
        //根据字节数组创建字符串
        byte[] byteArray = {23,34,56};
        String str3 = new String(byteArray);
        System.out.println("第三个字符串：" + str3);

    }
}

    2.引用类型使用==比较地址值

    /*
字符串常量池：程序中直接写上的双引号字符串，就在字符串常量池中

对于基本类型来说，==是进行数值的比较。
对于引用类型来说，==是进行[地址值]的比较。
*/
public class demo01Sting {
    public static void main(String[] args) {
        //定义两个字符串
        String str_1= "abc";
        String str_2= "abc";
        //使用char转换String的发生创建String
        char[] chararray = {'a','b','c'};
        String str_3 = new String(chararray);
        //因为是引用类型。所有他们比较的是地址值
        //因为直接创建的str_1和str_2在字符串常量池中
        //而str_3是由char转换成为的String，由于是引用类型使用比较的是地址值
        //两个byte大小虽然一样，但是储存空间不同，所以不相等
        System.out.println(str_1==str_2);//true
        System.out.println(str_1==str_3);//false
        System.out.println(str_2==str_3);//false
    }
}

    3.String当中获取相关的常用方法

    /*
String当中获取相关的常用方法有：
public int Length()：获取当前字符串长度
public String concat(String str)：当前字符串与参数字符串拼接成新的字符串
public char charAt(int index)：获取指定索引位置的单个字符，（索引从0开始）
public int indexOf(String str)：查找参数字符串在本字符串当中首次出现的索引位置，如果没有返回-1值
*/

public class demo01String {
    public static void main(String[] args) {
        //获取字符串的长度
        int length = "dfghjhfdsqwertgfdsa".length();
        System.out.println("字符串的长度是：" + length);
        //拼接字符串
        String str1 = "Hello";
        String str2 = "World";
        //把str2拼接在str1后面赋值新字符串str3
        String str3 = str1.concat(str2);
        System.out.println(str1);//原封不动
        System.out.println(str2);//原封不动
        System.out.println(str3);//HelloWorld    新字符串
        System.out.println("=========================");
        //获取指定索引位置的单个字符
        char ch ="Hello".charAt(1);
        System.out.println("第一号位置索引的字符是："+ch);
        //搜索对应字符串的索引位置
        String st = "qwe abc rtyhvcssertyjhvs abc";
        int index = st.indexOf("abc");
        System.out.println("第一次索引值是："+index);
    }
}

    4.字符串的截取方法

    /*
字符串的截取方法：
public String substring(int index)：截取从参数位置一直到字符串末尾，返回新字符串
public String substring(int begin，int end)：截取从begin开始，一直到end结束，中间的字符串
备注：[begin,end)  包含左边，不包含右边。
*/
public class demo01substring {
    public static void main(String[] args) {
        String str1 = "HelloWorld";
        //截取索引到结尾的全部
        String str2 = str1.substring(5);
        System.out.println(str1);
        System.out.println(str2);
        System.out.println("==============");
        //截取一部分
        String str3 = str1.substring(4,7);
        System.out.println(str3);
    }
}

    5.String转换常用方法

    /*
String当中与有：
public char[] toCharArray()：将当前字符串拆分成字符串数组作为返回值。
public byte[] getBytes()：获得当前字符串底层字节数组。
public String replace(CharSequence oldString, CharSequence newString)：
将所有出现的老字符串替换成为新的字符串，返回替换之后的结果新字符串。
*/
public class demo01 {
    public static void main(String[] args) {
        //转换成为字符数组,通过同CharArray方法
        char[] chars = "Hello".toCharArray();
        System.out.println(chars[0]);//H
        System.out.println(chars.length);//5
        System.out.println("===============");
        //把字符串转换为字节类型
        byte[] s = "abc".getBytes();
        for (int i = 0; i < s.length; i++) {
            System.out.println(s[i]);
        }
        String str = "你怎么玩的，你大爷的！";
        //通过replace方法把旧字符串替换为新的字符串
        System.out.println(str.replace("你大爷的","****"));
    }
}

    6.切割字符串

    /*
分割字符串的方法：
public String[] split(String regex)：按照参数的规则，将字符串拆分为若干部分
注意事项：
split方法的参数用的是"正则表达式"
所以英文句点“.”，进行切分。必须写"\\"
*/
public class demoStringsplit {
    public static void main(String[] args) {
        String str1 = "aaa,bbb,ccc";
        //字符串其实是一对引号里面是一个字符串
        System.out.println(str1);//aaa,bbb,ccc
        //使用split可以切割字符串，参数是选择切割的节点
        String[] split1 = str1.split(",");
        for (int i = 0; i < split1.length; i++) {
            System.out.println(split1[i]);//遍历出来的aaa bbb ccc
        }
        String str2 = "aaa.bbb.ccc";
        //因为是正则表达式，使用英文句点要使用"."
        String[] split2 = str2.split("\\.");
        for (int i = 0; i < split2.length; i++) {
            System.out.println(split2[i]);//遍历出来的aaa bbb ccc
        }
    }
}
